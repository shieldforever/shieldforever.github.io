#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<cstdlib>
#include<algorithm>
#include<cmath>
using namespace std;
void solve(){
    char s[110];
    scanf("%s", s + 1);
    int n = strlen(s + 1);
    int a[110], b[110];
    a[0] = 0; b[n + 1] = 0;
    for(int i = 1; i <= n; i++) a[i] = a[i - 1] + s[i] - '0';
    for(int i = n; i >= 1; i--) b[i] = b[i + 1] + s[i] - '0';
    int ans = 0;
    for(int i = 1; i <= n; i++){
        if(s[i] == '0'){
            if(a[i] > 0 && b[i] > 0) ++ans;
        }
    }
    printf("%d\n", ans);
    return ;
}
int main(){
    int T;
    cin>>T;
    while(T--){
        solve();
    }
    return 0;
}

#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<cstdlib>
#include<algorithm>
#include<cmath>
using namespace std;
typedef long long LL;
LL n, g, b;
void solve(){
    if(g >= (n + 1) / 2){
        printf("%lld\n", n); return ;
    }
    LL x = (n + 1) / 2;
    LL y = n - x;
    LL d = 0;
    if(g >= b){
        printf("%lld\n", n); return ;
    }
    LL ans;
    if(x % g != 0) ans = x / g * (g + b) + (x % g), y -= (x / g) * g, d = x / g;
    else ans = (x / g - 1) * (g + b) + g, y -= (x / g - 1) * g, d = (x / g - 1);
    d *= (b - g);
    if(d >= y) y = 0;
    else y -= d;
    printf("%lld\n", ans + y);
    return ; 
}
int main(){
    int T;
    cin>>T;
    while(T--){
        scanf("%lld%lld%lld", &n, &g, &b);
        solve();
    }
    return 0;
}

#include<iostream>

#include<cstring>
#include<string>
#include<cstdlib>
#include<algorithm>
#include<cmath>
using namespace std;
char s[220];
int g[27][27], v[27], e[27][3], ans[27];
void work(int x){
    v[x] = 1;
    ans[++ans[0]] = x;
    for(int i = 1; i <= e[x][0]; i++){
        if(!v[e[x][i]]) work(e[x][i]);
    }
    return ;
}
void solve(){
    int n = strlen(s + 1);
    memset(g, 0, sizeof(g));
    memset(e, 0, sizeof(e));
    memset(ans, 0, sizeof(ans));
    for(int i = 1; i < n; i++){
        int u = s[i] - 'a' + 1, v = s[i + 1] - 'a' + 1;
        g[u][v] = g[v][u] = 1;
    }
    for(int i = 1; i <= 26; i++){
        for(int j = 1; j <= 26; j++){
            if(g[i][j]) {
                if(e[i][0] + 1 >= 3){
                    printf("NO\n"); return ;
                }
                e[i][++e[i][0]] = j;
            }
        }
    }
    memset(v, 0, sizeof(v));
    for(int i = 1; i <= 26; i++){
        if(!v[i] && e[i][0] <= 1) work(i);
    }
    if(ans[0] != 26){
        printf("NO\n"); return ;
    }
    printf("YES\n");
    for(int i = 1; i <= 26; i++) printf("%c", ans[i] - 1 + 'a');
    printf("\n");
    return ;
}
int main(){
    int T;
    cin>>T;
    while(T--){
        scanf("%s", s + 1);
        solve();
    }
    return 0;
}

#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<cstdlib>
#include<algorithm>
#include<cmath>
using namespace std;
#define N 65
typedef long long LL;
LL n, m;
int b[N], num[N], len, maxl;
void work1(LL x){
    int i = -1;
    while(x > 0){
        ++i;
        if(x % 2 == 1) num[i] = 1;
        x /= 2;
    }
    len = i;
    return ;
}
void work2(LL a){
    int i = -1;
    while(a > 0) a /= 2, ++i;
    b[i]++;
    if(i > maxl) maxl = i;
    return ;
}
void init(){
    maxl = len = 0;
    memset(b, 0, sizeof(b));
    memset(num, 0, sizeof(num));
    scanf("%lld%lld", &n, &m);
    work1(n);
    for(int i = 1; i <= m; i++){
        LL a;
        scanf("%lld", &a);
        work2(a);
    }
    return ;
}
void solve(){
    int ans = 0;
    for(int i = 0; i <= len; i++){
        if(num[i] == 0) b[i + 1] += b[i] / 2;
        else{
            if(b[i] > 0) --b[i], b[i + 1] += b[i] / 2;
            else {
                int j;
                for(j = i; j <= maxl; j++) if(b[j] > 0) break;
                if(j > maxl){
                    printf("-1\n"); return ;
                }
                while(j != i){
                    --b[j]; ++ans;
                    b[j - 1] += 2;
                    --j;
                }
            }
        }
    }
    printf("%d\n", ans);
    return ;
}
int main(){
    int T;
    cin>>T;
    while(T--){
        init();
        solve();
    }
    return 0;
}
