#include<bits/stdc++.h>
using namespace std;
#define N 110
int a[N], n;
void solve(){
    scanf("%d", &n);
    int flag = 0;
    for(int i = 1; i <= n; i++){
        scanf("%d", &a[i]);
        if(a[i] % 2 == 0) flag = i;
    }
    if(!flag){
        if(n == 1)printf("-1\n");
        else printf("2\n1 2\n");
    }
    else{
        printf("1\n%d\n", flag);
    }
    return ;
}
int main(){
    int T;
    cin>>T;
    while(T--){
        solve();
    }
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
#define N 40040
typedef long long LL;
int n, m, k;
int a[N], b[N], s1[N], s2[N];
LL work(int l1, int l2){
    LL x = 0, y = 0;
    for(int i = 1; i + l1 -1 <= n; i++)
        if(s1[i + l1 - 1] - s1[i - 1] == l1) ++x;
    for(int i = 1; i + l2 -1 <= m; i++)
        if(s2[i + l2 - 1] - s2[i - 1] == l2) ++y;
    return x * y;
}
void solve(){
    LL ans = 0;
    for(int i = 1; i <= n; i++) s1[i] = s1[i - 1] + a[i];
    for(int i = 1; i <= m; i++) s2[i] = s2[i - 1] + b[i];
    int i;
    for(i = 1; i * i < k; i++){
        if(k % i == 0){
            ans += work(i, k / i);
            ans += work(k / i, i);
        }
    }
    if(i * i == k){
        ans += work(i, i);
    }
    printf("%lld\n", ans);
}
int main(){
    scanf("%d%d%d", &n, &m, &k);
    for(int i = 1; i <= n; i++) scanf("%d", &a[i]);
    for(int i = 1; i <= m; i++) scanf("%d", &b[i]);
    solve();
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
#define N 1000100
char s[N];
int n, l, r;
void init(){
    scanf("%d", &n);
    scanf("%s", s + 1);
    return ;
}
void solve(){
    int ans = 0;
    l = r = 0;
    int x = 0, y = 0;
    for(int i = 1; i <= n; i++){
        if(s[i] == '(') ++l;
        else ++r;
        if(x == 0){
            if(l < r) x = i;
        }
        else{
            if(l == r){
                y = i;
                ans += y - x + 1;
                x = 0; y = 0;
            }
        }
    }
    if(l == r) printf("%d\n", ans);
    else printf("-1\n");
    return ;
}
int main(){
    init();
    solve();
    return 0;
}
